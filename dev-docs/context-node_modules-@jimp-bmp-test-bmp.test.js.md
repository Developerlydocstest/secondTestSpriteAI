        0xff0080ff, 0xff00ffff, 0x8000ffff,
        0xff00ffff, 0x8000ffff, 0x0000ffff,
      ],
    });
    
    const buffer = await image.getBufferAsync("image/bmp");
    expect(buffer.toString()).toMatch(/^BMZ\u0000/);
  });
});
